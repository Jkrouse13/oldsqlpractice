Standard SQL Statements

SELECT

SELECT last_name,first_name
        FROM students
        WHERE student_id LIKE '%123%';

INSERT

INSERT INTO students VALUES
        ('Smith','John',123456789,'Math','Selleck');

INSERT SELECT

INSERT INTO students
        (student_id, first_name, last_name)
        SELECT student_id, first_name, last_name 
        FROM other_student_table;
        WHERE last_name LIKE '%son';

UPDATE

UPDATE students SET 
        last_name = 'Jones'
        WHERE student_id = 123;

DISTINCT

SELECT DISTINCT last_name
        FROM students;

ORDERING

SELECT last_name,first_name
        FROM students
        ORDER BY last_name, first_name DESC;

GROUP BY

SELECT last_name,COUNT(*)
        FROM students
        GROUP BY last_name;

LIKE

SELECT last_name
        FROM students
        WHERE last_name LIKE '%son';

COUNT

SELECT COUNT(*)
        FROM students
        WHERE status = 'Active';

COLUMN ALIASES

SELECT last_name,first_name AS first
        FROM students;

TABLE ALIASES

SELECT last_name, points
        FROM students S, assignments A 
        WHERE S.student_id = A.student_id AND 
        A.assignment_id = 12;

DELETE

DELETE FROM students 
        WHERE last_name = 'Smith';

CREATE TABLE

CREATE TABLE books (
  id              SERIAL PRIMARY KEY,
  title           VARCHAR(100) NOT NULL,
  primary_author  VARCHAR(100) NULL
);

DELETE TABLE

DROP TABLE students;

TRUNK TABLE

TRUNCATE TABLE students;

INSERT COLUMN

ALTER TABLE students 
        ADD COLUMN hometown varchar(20);

DELETE COLUMN

ALTER TABLE students 
        DROP COLUMN hometown;
